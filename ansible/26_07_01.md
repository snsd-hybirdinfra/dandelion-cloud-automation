# 25. 최종 backup.yml 구성

파일:


최종 작성 내용:

```yaml
---
- name: Backup 노드 Restic 백업 환경 구성
  hosts: backup
  become: true
  gather_facts: true

  tasks:
    - name: Backup 볼륨 마운트 상태 확인
      ansible.builtin.command:
        cmd: findmnt -n {{ backup_root }}
      register: backup_mount_check
      changed_when: false
      failed_when: false
      check_mode: false

    - name: Backup 볼륨 마운트 상태 검증
      ansible.builtin.assert:
        that:
          - backup_mount_check.rc == 0
        fail_msg: >-
          {{ backup_root }} 경로에 Backup Volume이 마운트되어 있지 않습니다.
          데이터 보호를 위해 Playbook 실행을 중단합니다.
        success_msg: >-
          {{ backup_root }} 경로의 Backup Volume 마운트를 확인했습니다.

    - name: Restic 백업 필수 패키지 설치
      ansible.builtin.apt:
        name:
          - restic
          - mariadb-client
          - rsync
          - openssh-client
          - cron
        state: present
        update_cache: true
        cache_valid_time: 3600

    - name: Root 전용 Backup 디렉터리 생성
      ansible.builtin.file:
        path: "{{ item }}"
        state: directory
        owner: root
        group: root
        mode: "0755"
      loop:
        - "{{ restic_repository }}"
        - "{{ restic_cache_dir }}"
        - "{{ backup_script_dir }}"

    - name: Backup 작업 및 결과 디렉터리 생성
      ansible.builtin.file:
        path: "{{ item }}"
        state: directory
        owner: ubuntu
        group: ubuntu
        mode: "0755"
      loop:
        - "{{ backup_staging_dir }}"
        - "{{ backup_staging_dir }}/db"
        - "{{ backup_staging_dir }}/wordpress"
        - "{{ backup_staging_dir }}/haproxy"
        - "{{ backup_staging_dir }}/prometheus"
        - "{{ backup_staging_dir }}/evidence"
        - "{{ backup_log_dir }}"
        - "{{ backup_restore_test_dir }}"

    - name: Restic 설정 디렉터리 생성
      ansible.builtin.file:
        path: /etc/restic
        state: directory
        owner: root
        group: root
        mode: "0700"

    - name: MariaDB 백업 설정 디렉터리 생성
      ansible.builtin.file:
        path: /etc/mysql/backup
        state: directory
        owner: root
        group: root
        mode: "0700"

    - name: Restic 환경 설정 파일 배포
      ansible.builtin.template:
        src: ../templates/backup/restic.env.j2
        dest: /etc/restic/restic.env
        owner: root
        group: root
        mode: "0644"

    - name: Restic Repository 비밀번호 파일 생성
      ansible.builtin.copy:
        dest: "{{ restic_password_file }}"
        content: "{{ restic_password }}\n"
        owner: root
        group: root
        mode: "0600"
      no_log: true

    - name: MariaDB 백업 계정 설정 파일 배포
      ansible.builtin.template:
        src: ../templates/backup/wp_backup.cnf.j2
        dest: "{{ backup_db_cnf_path }}"
        owner: root
        group: root
        mode: "0600"
      no_log: true

    - name: Restic 백업 스크립트 배포
      ansible.builtin.copy:
        src: ../files/backup/dandelion-restic-backup.sh
        dest: "{{ backup_script_path }}"
        owner: root
        group: root
        mode: "0755"

    - name: Restic 복구 테스트 스크립트 배포
      ansible.builtin.copy:
        src: ../files/backup/dandelion-restic-restore-test.sh
        dest: "{{ backup_restore_script_path }}"
        owner: root
        group: root
        mode: "0755"

    - name: Restic Repository 설정 파일 존재 여부 확인
      ansible.builtin.stat:
        path: "{{ restic_repository }}/config"
      register: restic_repository_config

    - name: Restic Repository 초기화
      ansible.builtin.command:
        cmd: restic init
      environment:
        RESTIC_REPOSITORY: "{{ restic_repository }}"
        RESTIC_PASSWORD_FILE: "{{ restic_password_file }}"
        RESTIC_CACHE_DIR: "{{ restic_cache_dir }}"
      when: not restic_repository_config.stat.exists
      no_log: true

    - name: Restic Repository 접근 확인
      ansible.builtin.command:
        cmd: restic snapshots
      environment:
        RESTIC_REPOSITORY: "{{ restic_repository }}"
        RESTIC_PASSWORD_FILE: "{{ restic_password_file }}"
        RESTIC_CACHE_DIR: "{{ restic_cache_dir }}"
      register: restic_snapshots_check
      changed_when: false
      check_mode: false

    - name: Restic Repository 상태 검증
      ansible.builtin.assert:
        that:
          - restic_snapshots_check.rc == 0
        fail_msg: "Restic Repository에 접근할 수 없습니다."
        success_msg: "Restic Repository 접근을 확인했습니다."

    - name: Cron 서비스 활성화 및 시작
      ansible.builtin.systemd_service:
        name: cron
        state: started
        enabled: true

    - name: 매일 Restic 자동 백업 cron 등록
      ansible.builtin.cron:
        name: "Dandelion Restic Backup"
        user: root
        minute: "{{ backup_cron_minute }}"
        hour: "{{ backup_cron_hour }}"
        job: "{{ backup_cron_job }}"
        state: present
```

# 29. 백업 스크립트 수동 실행 두 번째 오류


```text
dandelion-wp_wordpress
→ 0/2

→ 없음
```

이 문제는 Docker `--format` 표현을 사용하지 않거나 `jq`를 사용하여 우회하였다.
